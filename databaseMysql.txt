-- OPTIMIZED CAMPUSFLOW DATABASE SCHEMA
-- Drop existing database and recreate for clean schema
DROP DATABASE IF EXISTS campusflow;
CREATE DATABASE campusflow CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;
USE campusflow;

-- 1. USERS TABLE (Base authentication table)
CREATE TABLE users (
    id BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    email VARCHAR(100) NOT NULL UNIQUE,
    password VARCHAR(255) NOT NULL,
    name VARCHAR(100) NOT NULL,
    phone_number VARCHAR(15),
    role ENUM('TUTOR', 'STAFF', 'STUDENT', 'PARENT') NOT NULL,
    is_approved TINYINT(1) DEFAULT 0,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    INDEX idx_email (email),
    INDEX idx_role (role)
) ENGINE=InnoDB;

-- 2. STAFF TABLE
CREATE TABLE staff (
    staff_id VARCHAR(20) PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    department VARCHAR(50) NOT NULL,
    batch VARCHAR(20) NOT NULL,
    current_role ENUM('STAFF', 'TUTOR') NOT NULL,
    user_id BIGINT UNSIGNED UNIQUE,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE SET NULL,
    INDEX idx_department (department),
    INDEX idx_batch (batch)
) ENGINE=InnoDB;

-- 3. COURSES TABLE
CREATE TABLE courses (
    course_id VARCHAR(20) PRIMARY KEY,
    course_name VARCHAR(100) NOT NULL,
    course_code VARCHAR(20) NOT NULL UNIQUE,
    batch VARCHAR(20) NOT NULL,
    semester TINYINT UNSIGNED NOT NULL CHECK (semester BETWEEN 1 AND 8),
    credits TINYINT UNSIGNED DEFAULT 3,
    staff_id VARCHAR(20) NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (staff_id) REFERENCES staff(staff_id) ON DELETE RESTRICT,
    INDEX idx_batch_semester (batch, semester),
    INDEX idx_staff (staff_id)
) ENGINE=InnoDB;

-- 4. CLASSROOMS TABLE
CREATE TABLE classrooms (
    class_id VARCHAR(20) PRIMARY KEY,
    class_name VARCHAR(50) NOT NULL,
    tutor_id VARCHAR(20) NOT NULL,
    semester TINYINT UNSIGNED NOT NULL CHECK (semester BETWEEN 1 AND 8),
    batch VARCHAR(20) NOT NULL,
    department VARCHAR(50) NOT NULL,
    academic_year VARCHAR(20) NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (tutor_id) REFERENCES staff(staff_id) ON DELETE RESTRICT,
    INDEX idx_tutor (tutor_id),
    INDEX idx_batch_semester (batch, semester)
) ENGINE=InnoDB;

-- 5. SUBJECTS TABLE (Many-to-Many: Classrooms and Courses)
CREATE TABLE subjects (
    id BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    class_id VARCHAR(20) NOT NULL,
    course_id VARCHAR(20) NOT NULL,
    staff_id VARCHAR(20) NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (class_id) REFERENCES classrooms(class_id) ON DELETE CASCADE,
    FOREIGN KEY (course_id) REFERENCES courses(course_id) ON DELETE CASCADE,
    FOREIGN KEY (staff_id) REFERENCES staff(staff_id) ON DELETE RESTRICT,
    UNIQUE KEY unique_class_course (class_id, course_id),
    INDEX idx_class (class_id),
    INDEX idx_course (course_id)
) ENGINE=InnoDB;

-- 6. STUDENTS TABLE
CREATE TABLE students (
    register_number VARCHAR(20) PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    semester TINYINT UNSIGNED NOT NULL CHECK (semester BETWEEN 1 AND 8),
    class_id VARCHAR(20) NOT NULL,
    user_id BIGINT UNSIGNED UNIQUE,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (class_id) REFERENCES classrooms(class_id) ON DELETE RESTRICT,
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE SET NULL,
    INDEX idx_class (class_id),
    INDEX idx_semester (semester)
) ENGINE=InnoDB;

-- 7. PARENTS TABLE
CREATE TABLE parents (
    id BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    user_id BIGINT UNSIGNED NOT NULL UNIQUE,
    student_id VARCHAR(20) NOT NULL,
    relationship ENUM('FATHER', 'MOTHER', 'GUARDIAN') DEFAULT 'FATHER',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE,
    FOREIGN KEY (student_id) REFERENCES students(register_number) ON DELETE CASCADE,
    INDEX idx_student (student_id)
) ENGINE=InnoDB;

-- 8. ATTENDANCE TABLE
CREATE TABLE attendance (
    id BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    student_id VARCHAR(20) NOT NULL,
    subject_id BIGINT UNSIGNED NOT NULL,
    date DATE NOT NULL,
    class_hour TINYINT UNSIGNED NOT NULL CHECK (class_hour BETWEEN 1 AND 8),
    is_present TINYINT(1) NOT NULL DEFAULT 0,
    marked_by BIGINT UNSIGNED,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (student_id) REFERENCES students(register_number) ON DELETE CASCADE,
    FOREIGN KEY (subject_id) REFERENCES subjects(id) ON DELETE CASCADE,
    FOREIGN KEY (marked_by) REFERENCES users(id) ON DELETE SET NULL,
    UNIQUE KEY unique_attendance (student_id, subject_id, date, class_hour),
    INDEX idx_student_subject (student_id, subject_id),
    INDEX idx_date (date)
) ENGINE=InnoDB;

-- 9. EXAM_TYPES TABLE
CREATE TABLE exam_types (
    id INT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(50) NOT NULL UNIQUE,
    max_marks DECIMAL(5,2) NOT NULL,
    weightage DECIMAL(5,2),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
) ENGINE=InnoDB;

-- 10. MARKS TABLE
CREATE TABLE marks (
    id BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    student_id VARCHAR(20) NOT NULL,
    subject_id BIGINT UNSIGNED NOT NULL,
    exam_type_id INT UNSIGNED NOT NULL,
    marks_obtained DECIMAL(5,2) NOT NULL,
    max_marks DECIMAL(5,2) NOT NULL,
    entered_by BIGINT UNSIGNED,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    FOREIGN KEY (student_id) REFERENCES students(register_number) ON DELETE CASCADE,
    FOREIGN KEY (subject_id) REFERENCES subjects(id) ON DELETE CASCADE,
    FOREIGN KEY (exam_type_id) REFERENCES exam_types(id) ON DELETE RESTRICT,
    FOREIGN KEY (entered_by) REFERENCES users(id) ON DELETE SET NULL,
    UNIQUE KEY unique_student_subject_exam (student_id, subject_id, exam_type_id),
    INDEX idx_student (student_id),
    INDEX idx_subject (subject_id)
) ENGINE=InnoDB;

-- 11. QUERIES TABLE
CREATE TABLE queries (
    id BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    title VARCHAR(200) NOT NULL,
    message TEXT NOT NULL,
    from_user_id BIGINT UNSIGNED NOT NULL,
    to_user_id BIGINT UNSIGNED NOT NULL,
    subject_id BIGINT UNSIGNED,
    response TEXT,
    status ENUM('PENDING', 'RESPONDED', 'CLOSED') DEFAULT 'PENDING',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    responded_at TIMESTAMP NULL,
    FOREIGN KEY (from_user_id) REFERENCES users(id) ON DELETE CASCADE,
    FOREIGN KEY (to_user_id) REFERENCES users(id) ON DELETE CASCADE,
    FOREIGN KEY (subject_id) REFERENCES subjects(id) ON DELETE SET NULL,
    INDEX idx_from_user (from_user_id),
    INDEX idx_to_user (to_user_id),
    INDEX idx_status (status)
) ENGINE=InnoDB;

-- 12. COMPLAINTS TABLE
CREATE TABLE complaints (
    id BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    title VARCHAR(200) NOT NULL,
    description TEXT NOT NULL,
    from_user_id BIGINT UNSIGNED NOT NULL,
    about_student_id VARCHAR(20),
    class_id VARCHAR(20) NOT NULL,
    response TEXT,
    status ENUM('PENDING', 'RESPONDED', 'CLOSED') DEFAULT 'PENDING',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    responded_at TIMESTAMP NULL,
    FOREIGN KEY (from_user_id) REFERENCES users(id) ON DELETE CASCADE,
    FOREIGN KEY (about_student_id) REFERENCES students(register_number) ON DELETE SET NULL,
    FOREIGN KEY (class_id) REFERENCES classrooms(class_id) ON DELETE CASCADE,
    INDEX idx_from_user (from_user_id),
    INDEX idx_class (class_id),
    INDEX idx_status (status)
) ENGINE=InnoDB;

-- 13. GRIEVANCES TABLE
CREATE TABLE grievances (
    id BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    title VARCHAR(200) NOT NULL,
    description TEXT NOT NULL,
    from_user_id BIGINT UNSIGNED NOT NULL,
    to_user_id BIGINT UNSIGNED NOT NULL,
    subject_id BIGINT UNSIGNED,
    response TEXT,
    status ENUM('PENDING', 'IN_PROGRESS', 'RESOLVED', 'CLOSED') DEFAULT 'PENDING',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    resolved_at TIMESTAMP NULL,
    FOREIGN KEY (from_user_id) REFERENCES users(id) ON DELETE CASCADE,
    FOREIGN KEY (to_user_id) REFERENCES users(id) ON DELETE CASCADE,
    FOREIGN KEY (subject_id) REFERENCES subjects(id) ON DELETE SET NULL,
    INDEX idx_from_user (from_user_id),
    INDEX idx_to_user (to_user_id),
    INDEX idx_status (status)
) ENGINE=InnoDB;

-- 14. NOTIFICATIONS TABLE
CREATE TABLE notifications (
    id BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    title VARCHAR(200) NOT NULL,
    message TEXT,
    user_id BIGINT UNSIGNED NOT NULL,
    type ENUM('QUERY', 'GRIEVANCE', 'ANNOUNCEMENT', 'COMPLAINT', 'GENERAL') DEFAULT 'GENERAL',
    reference_id BIGINT UNSIGNED,
    is_read TINYINT(1) DEFAULT 0,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE,
    INDEX idx_user_read (user_id, is_read),
    INDEX idx_created (created_at)
) ENGINE=InnoDB;

-- 15. ANNOUNCEMENTS TABLE
CREATE TABLE announcements (
    id BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    title VARCHAR(200) NOT NULL,
    content TEXT NOT NULL,
    posted_by BIGINT UNSIGNED NOT NULL,
    class_id VARCHAR(20) NOT NULL,
    is_active TINYINT(1) DEFAULT 1,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    FOREIGN KEY (posted_by) REFERENCES users(id) ON DELETE CASCADE,
    FOREIGN KEY (class_id) REFERENCES classrooms(class_id) ON DELETE CASCADE,
    INDEX idx_class_active (class_id, is_active),
    INDEX idx_created (created_at)
) ENGINE=InnoDB;
